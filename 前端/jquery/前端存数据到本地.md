// 函数中的参数为 要获取的cookie键的名称。
function getCookie(userName){
    if (document.cookie.length>0){
        c_start=document.cookie.indexOf(userName+ "=");
        if (c_start!=-1){
            c_start=c_start + userName.length+1;
            c_end=document.cookie.indexOf(";",c_start);
            if (c_end==-1){ 
                c_end=document.cookie.length;
            }
            return unescape(document.cookie.substring(c_start,c_end));
        }
    }
    return "";
}

#### 3. 对设置的数据进行操作

```js
var MyLocalStorage = {
  Cache: {
    put: function put(key, value, time) {
      try {
        if (!localStorage) {
          return false
        }
        if (!time || isNaN(time)) {
          time = 60
        }
        var cacheExpireDate = new Date().getTime() + time * 1000;
        var cacheVal = { val: value, exp: cacheExpireDate }
        localStorage.setItem(key, JSON.stringify(cacheVal))
      } catch (e) {
        console.log(e)
      }
    },
    get: function get(key) {
      try {
        if (!localStorage) {
          return false
        }

        var cacheVal = localStorage.getItem(key)
        var result = JSON.parse(cacheVal)

        var now = new Date().getTime()
        if (!result) {
          return null
        }
        if (now > result.exp) {
          this.remove(key)
          return ""
        }
        return result.val
      } catch (e) {
        this.remove(key)
        return null
      }
    },
    remove: function remove(key) {
      if (!localStorage) {
        return false
      }
      localStorage.removeItem(key)
    },
    clear: function clear() {
      if (!localStorage) {
        return false;
      }
      localStorage.clear()
    }
  }
}
```
